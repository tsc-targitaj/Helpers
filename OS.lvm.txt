// посчитать использование LV при thin lvm

VG="vgssd"; THIN_POOL="lvm-ssd"; LVM_SSD=$(lvs $VG | awk -v pool="$THIN_POOL" '$1 == pool {print $4}'); echo "$THIN_POOL total size is:   $LVM_SSD"; lvs $VG --units g --noheadings -o lv_size | grep -v $LVM_SSD | awk -v pool="$THIN_POOL" '{sum += $1} END {print pool " сумма LV в пуле:", sum "G"}'
